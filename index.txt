function getLibyaSecsLeft(){
  const now=new Date();
  const libyaMs=now.getTime()+(now.getTimezoneOffset()*60000)+(2*3600000);
  const d=new Date(libyaMs);
  return 86400-(d.getHours()*3600+d.getMinutes()*60+d.getSeconds());
}
function tick(){
  const s=getLibyaSecsLeft();
  document.getElementById('timer').textContent=`${String(Math.floor(s/3600)).padStart(2,'0')}:${String(Math.floor((s%3600)/60)).padStart(2,'0')}:${String(s%60).padStart(2,'0')}`;
}

  const list=pricesDiv.querySelector('.price-list');
  list.innerHTML='';
  pricesDiv.classList.add('open');
  if(!country.prices||country.prices.length===0){
    list.innerHTML='<p style="font-size:12px;color:var(--muted);text-align:center;padding:10px;">لا توجد فئات لهذه الدولة</p>';
    return;
  }
  country.prices.forEach(item=>{
    const b=document.createElement('button');
    b.className='price-btn';
    b.innerHTML=`<span>${product.name} ${item.label} (${country.flag} ${country.name})</span><span class="lyd">${item.price} د.ل <span class="add-icon">+</span></span>`;
    b.onclick=()=>addToCart(`${product.name} ${item.label} (${country.flag} ${country.name})`,item.price,pid);
    list.appendChild(b);
  });
}

/* ===== IMG VALIDATION ===== */
function validateImgUrl(inputId,statusId){
  const val=document.getElementById(inputId).value.trim();
  const status=document.getElementById(statusId);
  if(!val){status.textContent='';status.className='img-url-status';return;}
  try{
    const url=new URL(val);
    if(!['http:','https:'].includes(url.protocol)){status.textContent='⚠️ يجب أن يبدأ بـ https://';status.className='img-url-status err';return;}
    const img=new Image();
    img.onload=()=>{status.textContent='✅ الصورة صالحة';status.className='img-url-status ok';};
    img.onerror=()=>{status.textContent='⚠️ تعذّر تحميل الصورة';status.className='img-url-status err';};
    img.src=val; status.textContent='⏳ جاري التحقق...'; status.className='img-url-status';
  }catch{status.textContent='⚠️ رابط غير صحيح';status.className='img-url-status err';}
}

/* ===== CART ===== */
function toggleCart(){
  const p=document.getElementById('cart-panel');
  p.style.display=p.style.display==='block'?'none':'block';
}
function addToCart(name,unitPrice,productId){
  const existing=cart.find(i=>i.name===name&&i.unitPrice===unitPrice);
  if(existing){existing.qty++;existing.price=existing.unitPrice*existing.qty;}
  else{cart.push({name,unitPrice,price:unitPrice,qty:1,productId});}
  saveCart();renderCart();
  showToast(`✅ تمت الإضافة: ${name}`);
  document.getElementById('cart-panel').style.display='block';
}
function changeQty(i,delta){
  if(!cart[i])return;
  cart[i].qty+=delta;
  if(cart[i].qty<=0){cart.splice(i,1);}
  else{cart[i].price=cart[i].unitPrice*cart[i].qty;}
  saveCart();renderCart();
}
function removeFromCart(i){cart.splice(i,1);saveCart();renderCart();}
function clearCart(){
  if(!cart.length){showToast('⚠️ السلة فارغة');return;}
  cart=[];saveCart();renderCart();showToast('🗑 تم إفراغ السلة');
}
function renderCart(){
  const itemsEl=document.getElementById('cart-items');
  const totalEl=document.getElementById('total');
  const countEl=document.getElementById('cart-count');
  const totalItems=cart.reduce((s,i)=>s+i.qty,0);
  if(!cart.length){
    itemsEl.innerHTML='<p style="color:var(--muted);font-size:13px;text-align:center;padding:10px 0;">السلة فارغة</p>';
    countEl.style.display='none';
  } else {
    itemsEl.innerHTML=cart.map((item,i)=>`
      <div class="cart-item-row">
        <span class="name">${item.name}</span>
        <div class="qty-ctrl">
          <button class="qty-btn" onclick="changeQty(${i},-1)">−</button>
          <span class="qty-num">${item.qty}</span>
          <button class="qty-btn" onclick="changeQty(${i},1)">+</button>
        </div>
        <span class="price">${item.price} د.ل</span>
        <button class="rm" onclick="removeFromCart(${i})">✕</button>
      </div>`).join('');
    countEl.textContent=totalItems; countEl.style.display='flex';
  }
  totalEl.textContent=cart.reduce((s,i)=>s+i.price,0);
}
renderCart();

function initiateCheckout(){
  if(!cart.length){showToast('⚠️ السلة فارغة!');return;}
  const enabled=state.paymentMethods.filter(m=>m.enabled);
  if(!enabled.length){showToast('⚠️ لا توجد طرق دفع مفعّلة');return;}
  const baseTotal=cart.reduce((s,i)=>s+i.price,0);
  const list=document.getElementById('payment-methods-list');
  list.innerHTML=enabled.map(pm=>{
    const finalTotal=calcTotal(baseTotal,pm.id);
    const feeText=pm.feePercent>0?` (+${pm.feePercent}%)`:'';
    return `<button class="payment-method-btn" onclick="selectPayment('${pm.id}',this)">
      <span class="pm-icon">${pm.icon}</span>
      <span class="pm-info">
        <div class="pm-name">${pm.name}</div>
        <div class="pm-detail">${pm.account}${feeText}</div>
      </span>
      <span class="pm-price">${finalTotal} د.ل</span>
    </button>`;
  }).join('');
  document.getElementById('payment-modal').classList.add('show');
}
